数据库语句总结

Sep 21, 2019

模式的定义与删除

定义

create schema schemaName

删除

drop schema schemaName

表的定义与删除

定义

create table tableName(
columnName1 dataType1 restrict1,
columnName2 dataType2 restrict2
)tableRestrict;

修改基本表

alter table tableName
add column columnName datatype resctirct
drop column columnName
drop constriant restrictName
alter column columnName datatype

删除基本表

drop table tableName

索引的建立与删除

建立索引

create index indexName
on tableName(columnName)

修改索引

alter index oldIndexName
rename to newIndexName

删除索引

drop index indexName

数据查询

增 insert

insert into tableName(fieldName1,fieldName2,…)
values (value1,value2,…)

删 delete

delete from tableName
where columnName1=”value”

改 update

update tableName
set fieldName1=”value1”
where fieldName2=”value2”

查 select

select fieldName1,fieldName2,
from tableName
where fieldName3=”value3”
group by
order by

表连接 join

##
select tableName1.fieldName1
from
tableName1 left join tableName2
on tableName1.id=tableName2.id
